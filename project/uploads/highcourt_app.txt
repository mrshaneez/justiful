function timeToMins(t){ const [h,m]=t.split(":").map(Number); return h*60+m; }
function timesOverlap(s1,e1,s2,e2){ return timeToMins(s1)<timeToMins(e2)&&timeToMins(e1)>timeToMins(s2); }

function detectClashes(bookings, newB, excludeId=null) {
  const clashes = [];
  const others = bookings.filter(b=>b.id!==excludeId&&b.date===newB.date&&b.status!=="Cancelled");
  for (const b of others) {
    if (!timesOverlap(newB.timeStart,newB.timeEnd,b.timeStart,b.timeEnd)) continue;
    if (b.courtroom===newB.courtroom) clashes.push({ type:"courtroom", msg:`${b.courtroom} already booked for "${b.caseTitle}" (${b.timeStart}–${b.timeEnd})` });
    const jC = newB.judges.filter(j=>b.judges.includes(j));
    if (jC.length) clashes.push({ type:"judge", msg:`${jC.map(j=>j.replace("Hon. ","")).join(", ")} already scheduled in "${b.caseTitle}"` });
    const lC = newB.lawyers.filter(l=>b.lawyers.includes(l));
    if (lC.length) clashes.push({ type:"lawyer", msg:`${lC.join(", ")} already scheduled in "${b.caseTitle}"` });
    const pC = newB.parties.filter(p=>b.parties.includes(p));
    if (pC.length) clashes.push({ type:"party", msg:`"${pC.join(", ")}" already in "${b.caseTitle}"` });
  }
  return clashes;
}

      <div style={{marginBottom:12}}><div style={{fontSize:12,fontWeight:500,marginBottom:6}}>Counsel</div><div style={{display:"flex",gap:8,flexWrap:"wrap"}}>{allLawyers.map(l=><label key={l} style={{display:"flex",alignItems:"center",gap:5,fontSize:12,padding:"4px 10px",borderRadius:99,background:f.lawyers.includes(l)?S.accentLight:S.surface2,border:`0.5px solid ${f.lawyers.includes(l)?S.accentMid:S.border}`,cursor:"pointer",color:f.lawyers.includes(l)?S.accent:S.text}}><input type="checkbox" checked={f.lawyers.includes(l)} onChange={()=>toggleArr("lawyers",l)} style={{display:"none"}}/>{l}</label>)}</div></div>
      <Textarea label="Notes" value={f.notes} onChange={e=>upd("notes",e.target.value)} rows={2}/>
      {checked&&(clashes.length===0?<div style={{marginBottom:12,background:S.success,color:S.successT,padding:"10px 14px",borderRadius:8,fontSize:13}}>No conflicts detected. All clear.</div>:<div style={{marginBottom:12,background:S.danger,color:S.dangerT,padding:"10px 14px",borderRadius:8,fontSize:13}}><div style={{fontWeight:500,marginBottom:6}}>{clashes.length} conflict{clashes.length>1?"s":""} detected:</div>{clashes.map((c,i)=><div key={i} style={{display:"flex",gap:6,marginBottom:3,alignItems:"flex-start"}}><span style={{fontSize:10,marginTop:2,background:S.dangerT,color:S.danger,padding:"1px 5px",borderRadius:99,textTransform:"uppercase",flexShrink:0}}>{c.type}</span><span style={{fontSize:12}}>{c.msg}</span></div>)}</div>)}
      <div style={{display:"flex",gap:8,justifyContent:"space-between",alignItems:"center",paddingTop:4}}>
        <Btn variant="ghost" onClick={checkClashes}>Check for clashes</Btn>
        <div style={{display:"flex",gap:8}}><Btn variant="ghost" onClick={onClose}>Cancel</Btn><Btn variant="primary" onClick={save}>{isEdit?"Save changes":"Confirm booking"}</Btn></div>
      </div>
    </Modal>
  );
}
